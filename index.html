import random def play_game(): options = ["剪刀", "石頭", "布"] print("--- 歡迎來到野球拳遊戲！ ---") print("請輸入：剪刀、石頭 或 布（輸入 '離開' 結束遊戲）") while True: # 玩家輸入 user_choice = input("\n你的選擇: ").strip() if user_choice == "離開": print("遊戲結束，下次再玩！") break if user_choice not in options: print("無效的輸入，請重新輸入：剪刀、石頭 或 布。") continue # 電腦隨機選擇 computer_choice = random.choice(options) print(f"電腦出的是: {computer_choice}") # 勝負邏輯判定 if user_choice == computer_choice: print("結果：平手！再試一次。") elif (user_choice == "石頭" and computer_choice == "剪刀") or \ (user_choice == "剪刀" and computer_choice == "布") or \ (user_choice == "布" and computer_choice == "石頭"): print("結果：你贏了！太棒了！") else: print("結果：你輸了，再接再厲！") if __name__ == "__main__": play_game()
